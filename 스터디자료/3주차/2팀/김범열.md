## 스위치 실습
### 코드
int led = 13;
int sw =2;

void setup() { 
  pinMode(led, OUTPUT); //LED기 츨력
  pinMode(sw, INPUT); //스위치가 입력
  }

void loop() {
  if (digitalRead(sw) == HIGH) { //스위치가 하이면
    digitalWrite(led, HIGH);  //LED가 하이로 변환 후 켜진다 
  }
  else{
      digitalWrite(led, LOW); //아니면 LED가 로우로 변환 꺼진다
  }
}
